**قبل:**
```python
context_parts.append(f"سند {i}: {result['text'][:400]}...")
```

**بعد:**
```python
metadata = result.get('metadata', {})

# ساخت context با metadata کامل
doc_context = f"سند {i}:\n"

# اضافه کردن metadata مهم (اگر موجود باشد)
if metadata.get('hierarchy_code'):
    doc_context += f"   📌 کد طبقه‌بندی: {metadata.get('hierarchy_code')}\n"
if metadata.get('hierarchy_title'):
    doc_context += f"   📄 عنوان: {metadata.get('hierarchy_title')}\n"
if metadata.get('parent_clause'):
    doc_context += f"   🔗 بند والد: {metadata.get('parent_clause')}\n"

    doc_context += f"   🔗 بخش والد: {metadata.get('parent_section')}\n"

# اضافه کردن محتوای اصلی
```
