```sql
SELECT SUM(COALESCE(CAST("جمع_در_آمد_اختصاصي" AS DOUBLE PRECISION), 0)) AS total_amount 
FROM incomes_sheet1 
WHERE (
  (TRANSLATE("عنوان_دستگاه", 'يكيۀةأإٱآ', 'یکیهههااا') ILIKE '%انستیتو%' 
   AND TRANSLATE("عنوان_دستگاه", 'يكيۀةأإٱآ', 'یکیهههااا') ILIKE '%پاستور%' 
   AND TRANSLATE("عنوان_دستگاه", 'يكيۀةأإٱآ', 'یکیهههااا') ILIKE '%ایران%')
  OR 
  (TRANSLATE("عنوان_دستگاه_اصلی", 'يكiۀةأإٱآ', 'یکیهههااا') ILIKE '%انستیتو%' 
   AND TRANSLATE("عنوان_دستگاه_اصلی", 'يكiۀةأإٱآ', 'یکیهههااا') ILIKE '%پاستور%' 
   AND TRANSLATE("عنوان_دستگاه_اصلی", 'يكiۀةأإٱآ', 'یکیهههااا') ILIKE '%ایران%')

AND TRANSLATE("سال", 'يكيۀة', 'یکیهه') IN ('1401', '1402', '1403')
```
